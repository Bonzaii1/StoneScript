//DEADWOOD SCRIPT
// -Bezerra, the Sage
func speak(message)
    >o-7,0,#red,@message@

func setBase()
    equipL sword *8
    equipR compound shield *8

var run = 0
var fastest_run = 0

?loc=Deadwood
    >`0,1,Current Time = @time.FormatCasual(totaltime)@
    >`65,0,Fastest Run = @time.FormatCasual(loc.bestTime)@ 
    ?run = 0
        >`0,0,Prev Run = First Run :p
    :
        >`0,0,Prev Run = @time.FormatCasual(run)@
    ?time <=1
        setBase()
    ?time >= 0 & time <= (30*5)
        speak("OKKKKKKK, HERE WE GOOOOOO")
    
    ?foe

        ?foe = tree_boss
            ?ai.paused
                speak("Bruh, hurry it up man.")


            ?foe.distance > 5
                equipR dashing shield
            :
                foe.state = 32
                    equipL poison wand *7
                    equipR compound shield *8
                :
                    setBase()
        :
            ?foe.count > 2
                ?foe.distance > 10
                    equipR dashing shield
                :
                    equip bardiche

            :
                ?foe.distance > 5
                    equipR dashing shield
                :
                    setBase()
    :?pickup
        equipL star
        equipR compound shield *8
    :
        setBase()
    
    ?ai.paused & foe.count = 0
        run = totaltime
        ?fastest_run = 0 | run < fastest_run
            fastest_run = run
